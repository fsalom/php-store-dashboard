		input.text { margin-bottom:12px; width:95%; padding: .4em; 	outline: none; }
		fieldset { padding:0; border:0; margin-top:25px; }
		.ui-dialog .ui-state-error { padding: .3em; }
		.validateTips { border: 1px solid transparent; padding: 0.3em; }
		ui-dialog{
			background-color:#990000;
		}
	</style>
	
	<script>
	$(function() {
		// a workaround for a flaw in the demo system (http://dev.jqueryui.com/ticket/4375), ignore!
		$( "#dialog:ui-dialog" ).dialog( "destroy" );
		
		var name = $( "#name" ),
			price = $( "#price" ),
			allFields = $( [] ).add( name ),
			tips = $( ".validateTips" );

		function updateTips( t ) {
			tips
				.text( t )
				.addClass( "ui-state-highlight" );
			setTimeout(function() {
				tips.removeClass( "ui-state-highlight", 1500 );
			}, 500 );
		}

		function checkLength( o, n, min, max ) {
			if ( o.val().length > max || o.val().length < min ) {
				o.addClass( "ui-state-error" );
				updateTips( "La longitud del texto debe de ser entre " +
					min + " y " + max + "." );
				return false;
			} else {
				return true;
			}
		}

		function checkRegexp( o, regexp, n ) {
			if ( !( regexp.test( o.val() ) ) ) {
				o.addClass( "ui-state-error" );
				updateTips( n );
				return false;
			} else {
				return true;
			}
		}
		
		function isNumber(n) {
			value=n.val();
  			//!isNaN(parseFloat(value)) && isFinite(value);
  			if((!isNaN(parseFloat(value))) && (isFinite(value))){
  				return true;
  			}else{
  				updateTips( "Debe de ser un número. Los decimales deben señalarse con un . ejem: (0.12)" );
  				n.addClass( "ui-state-error" );
				return false;	
  			}
		}
		
		$( "#dialog-form" ).dialog({
			autoOpen: false,
			height: 350,
			width: 350,
			modal: true,
			buttons: {
				"Crear una nueva acción": function() {
					var bValid = true;
					allFields.removeClass( "ui-state-error" );

					bValid = bValid && checkLength( name, "name", 3, 30 );
					nValid = isNumber(price);

					
					if ( bValid && nValid) {
						
						$.post("actions.php", {
												name:name.val(), 
												type:$("#type").val(), 
												price:$("#price").val(), 
												id:$("#id").val(), 
												item:$("#item").val()
												},
												 function(data) {
                        	$( "#myTable tbody" ).append( "<tr>" +
							"<td>" + data.name + "</td>" +
							"<td>" + data.type +"</td>" +
							"<td>" + data.price +"</td>" + 
							'<td> <a href="'+ data.url +'"  class="button red" onclick="return confirmar()">X</a></td>' +
						"</tr>" ); 
                        	
                        	//window.location.href ="?go=item&id_item=" + id;
                        
                    	}, "json");
                    	
						$( this ).dialog( "close" );
					}
					
                   
				},
				Cancel: function() {
					$( this ).dialog( "close" );
					//window.location.href ="?go=item&id_item=" + id;
				}
			},
			close: function() {
				allFields.val( "" ).removeClass( "ui-state-error" );
				//window.location.href ="?go=item&id_item=" + id;
			}
		});

		$( "#create-user" )
			.button()
			.click(function() {
				$( "#dialog-form" ).dialog( "open" );
			});
	});
	</script>

<script type="text/javascript">

        function confirmar( )
        {
            if(confirm( "Estas seguro que deseas eliminar el registro?"  ))
                return true;
            else
                return false;
        }

</script>
